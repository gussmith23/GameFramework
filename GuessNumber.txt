init:
    number_to_guess is `randint(0, 10)`.
    guessed_num is -1.

    state is choose.

    if guessed_num is number_to_guess:
        state is victory.
    else:
        state is wrong.

wrong:
    Say "You guessed wrong".
    state is choose.

victory:
    Say "Good job!".
    Finish.
