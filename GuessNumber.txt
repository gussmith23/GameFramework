state init:
    guessed_num is -1.
    number_to_guess is `randint(0,10)`.
    Say "Welcome to guess a number!".
    next_state is "choose".

state choose:
    Get guessed_num.
    if guessed_num is number_to_guess:
        next_state is "victory".
    else:
        next_state is "wrong".

state wrong:
    Say "You guessed wrong".
    next_state is "choose".

state victory:
    Say "Good job!".
    next_state is "finished".

state finished:
    Say "something".
    Finish.
