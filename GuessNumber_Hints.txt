(This version of guess the number will give hints)

fields = {
    'number_to_guess'   : randint(0, 10), 
    'guessed_num'       : -1,
    'state'             : "init"
}

init:
    Say "Welcome to guess a number!".
    state is choose.

choose:
    Get guessed_num.
    if guessed_num is number_to_guess:
        state is victory
    else:
        state is wrong.

wrong:
    if guessed_num less than number_to_guess: Say "You guessed too low",
    else: Say "You guessed too high".
    state is choose.

victory:
    Say "Good job!".
    Finish.
