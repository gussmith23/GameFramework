(This version of guess the number will give hints)

init:
    guessed_num is -1.
    number_to_guess is `randint(0,10)`.
    Say "Welcome to guess a number!".
    next_state is "choose".

state choose:
    Say "Enter a number: "
    Get guessed_num.
    if guessed_num is number_to_guess:
        next_state is "victory".
    else:
        next_state is "wrong".

wrong:
    if guessed_num < number_to_guess: Say "You guessed too low",
    else: Say "You guessed too high".
    next_state is choose.

victory:
    Say "Good job!".
    Finish.
